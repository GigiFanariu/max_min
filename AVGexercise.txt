CREATE VIEW MEAL_ORDER_WITH_DETAILS AS 
    SELECT MO.*, M.PRICE, M.CALORIES FROM MEAL_ORDER MO, MEAL M 
    WHERE MO.MEAL_NAME = M.NAME;

SELECT MAX(PRICE) AS MAX_PRICE
FROM MEAL
WHERE CALORIES > 500;

SELECT AVG(CALORIES) AS AVG_CALORIES
FROM MEAL
WHERE PRICE > 15.0;

SELECT * FROM MEAL
WHERE PRICE = (SELECT MAX(PRICE) FROM MEAL);

SELECT * FROM MEAL
HAVING PRICE < (SELECT AVG(PRICE) FROM MEAL);

SELECT * FROM MEAL
